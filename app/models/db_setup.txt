CREATE DATABASE POOLCAR;

CREATE TABLE POOLCAR.PROFILE(
	LOGIN VARCHAR(32),
    PASSWORD VARCHAR(32),
	PHONE_NUMBER VARCHAR(32),
    NAME VARCHAR(32),
    SURNAME VARCHAR(32),
    EMAIL VARCHAR(32),
    AGE VARCHAR(32),
    COUNTRY VARCHAR(32),
    CONSTRAINT PK_PROFILE PRIMARY KEY (LOGIN)
);

CREATE TABLE POOLCAR.CAR (
	CAR_BRAND VARCHAR(32),
	SEATS VARCHAR(4),
    COLOUR VARCHAR(32),
    CONSTRAINT PK_CAR PRIMARY KEY (CAR_BRAND, SEATS, COLOUR)
);
