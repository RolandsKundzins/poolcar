CREATE TABLE POOLCAR.PROFILES(
    USER_ID VARCHAR(32),
	LOGIN VARCHAR(32),
    PASSWORD VARCHAR(32),
	PHONE_NUMBER VARCHAR(32),
    NAME VARCHAR(32),
    SURNAME VARCHAR(32),
    EMAIL VARCHAR(32),
    AGE VARCHAR(32),
    COUNTRY VARCHAR(32),
    SCORE VARCHAR(4),
    SCORE_DIVIDED_BY VARCHAR(32),
    CONSTRAINT PK_PROFILE PRIMARY KEY (USER_ID)
);
